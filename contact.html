const form = document.querySelector("#form") const submitButton = document.querySelector("#submit") const scriptURL = '' // google sheets url form.addEventListener('submit', e => { submitButton.disabled = true e.preventDefault() let requestBody = new FormData(form) fetch(scriptURL, { method: 'POST', body: requestBody}) .then(response => { alert('Message send Successfully', response) submitButton.disabled = false }) .catch(error => { alert('Error sending message', error.message) submitButton.disabled = false } ) })
